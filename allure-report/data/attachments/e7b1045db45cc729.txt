pm.test("Status code is 500", function () {
    pm.response.to.have.status(500);
});

pm.test("A resposta é valida e deve ter um corpo", function () {
     pm.response.to.be.withBody;
     pm.response.to.be.json;
});

// Teste para verificar a resposta da requisição
pm.test("Validar resposta de erro", function () {
    // Extrair o corpo da resposta em JSON
    var responseBody = pm.response.json();
       // Verificar o valor do campo "errorCode"
    pm.expect(responseBody.error.errorCode).to.equal("9999");

    // Verificar o valor do campo "message"
    pm.expect(responseBody.error.message).to.equal("Internal Server Error");
});
