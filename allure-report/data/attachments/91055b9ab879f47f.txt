pm.test("Status code is 404", function () {
    pm.response.to.have.status(404);
});

pm.test("A resposta é valida e deve ter um corpo", function () {
    pm.response.to.be.withBody;
    pm.response.to.be.json;
});

pm.test("Validar mensagem de erro", function () {
  pm.response.to.have.jsonBody("error.message", "Cliente não encontrado");
  pm.response.to.have.jsonBody("error.errors[0].message", "documentNumber informado não encontrado");
});
