pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

// Obtém o valor da variável "token" da resposta JSON
var jsonData = pm.response.json();
var token = jsonData.access_token;

// Define a variável "token" como global
pm.globals.set("token", token);

// Imprime o valor da variável "token" no console
console.log("Valor da variável global token: " + token);
