pm.test("Status code is 422", function () {
    pm.response.to.have.status(422);
});

pm.test("A resposta é valida e deve ter um corpo", function () {
    pm.response.to.be.withBody;
    pm.response.to.be.json;
});

// Extrai a mensagem de erro da resposta JSON
var errorMessage = pm.response.json().error.message;

// Define o valor esperado
var expectedErrorMessage = "ECMA 262 regex";

// Verifica se a mensagem de erro contém o valor esperado
pm.test("Validating error message", function () {
    pm.expect(errorMessage).to.include(expectedErrorMessage);
