    pm.test("Validar o formato do campo email", function () {
        var data = pm.response.json();
        pm.expect(data.email).to.match(/^[\w-\.]+@([\w-]+\.)+[\w-]{2,4}$/);

    });

    pm.test("Validar a comparação idPessoa entre Request e Response", function () {
       var url = pm.request.url.toString();
       var value = url.split("/");
       for(var x=0; x<=value.length-1; x++){
           if(value[x].match(value[x], /^[0-9]+$/)){
               var requestIdUrl = value[x].toString();
           }
       }
       var respondeIdPessoa = pm.response.json();
       pm.expect(respondeIdPessoa.idPessoa).to.equal(parseInt(requestIdUrl));
       
    });


    pm.test("GET Status code is 200", function () {
        pm.response.to.have.status(200);
    });

    pm.test("A resposta é valida e deve ter um corpo", function () {
        pm.response.to.be.ok;
        pm.response.to.be.withBody;
        pm.response.to.be.json;
