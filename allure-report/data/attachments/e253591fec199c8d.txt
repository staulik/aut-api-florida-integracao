pm.test("Status code is 500", function () {
    pm.response.to.have.status(500);
});

pm.test("A resposta é valida e deve ter um corpo", function () {
     pm.expect(pm.response.text()).to.include("Internal Server Error");
     pm.response.to.be.withBody;
     pm.response.to.be.json;
});

// Verificar se a resposta contém a mensagem de erro esperada
pm.test("Validar mensagem de erro", function () {
    var jsonData = pm.response.json();
    pm.expect(jsonData.error.message).to.equal("Internal Server Error");
    pm.expect(jsonData.error.errorCode).to.equal("9999");
});
